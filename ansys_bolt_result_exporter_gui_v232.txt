# Bolt ENFO/ENMO result exporter with a Windows Forms GUI.
# Exports wide hand-review tables, tidy postprocessing tables, manifest data,

# It expects bolt UDR folders produced with names like:
#   Bolt Results/ENFOX/ENFOX_NS_BoltNode_001

EXPRESSIONS = ["ENFOX", "ENFOY", "ENFOZ", "ENMOX", "ENMOY", "ENMOZ"]
FORCE_EXPRESSIONS = ["ENFOX", "ENFOY", "ENFOZ"]
MOMENT_EXPRESSIONS = ["ENMOX", "ENMOY", "ENMOZ"]
DERIVED_QUANTITIES = [
    "ForceTotal",
    "MomentTotal",
    "ShearIfXAxial",
    "ShearIfYAxial",
    "ShearIfZAxial",
    "MomentShearIfXAxial",
    "MomentShearIfYAxial",
    "MomentShearIfZAxial",
]
WIDE_VALUE_COLUMNS = EXPRESSIONS + DERIVED_QUANTITIES
INDIVIDUAL_ROOT = "Bolt Results"
SCREENING_ROOT = "Bolt Results - Max/Min Screening"
SCREENING_LABELS = [
    "MaximumOverTime",
    "TimeOfMaximum",
    "MinimumOverTime",
    "TimeOfMinimum",
]
SCREENING_DRIVER_TO_LABEL = {
    "MaximumOverTime": "MaximumOverTime",
    "Maximum Over Time": "MaximumOverTime",
    "TimeOfMaximum": "TimeOfMaximum",
    "Time Of Maximum": "TimeOfMaximum",
    "MinimumOverTime": "MinimumOverTime",
    "Minimum Over Time": "MinimumOverTime",
    "TimeOfMinimum": "TimeOfMinimum",
    "Time Of Minimum": "TimeOfMinimum",
}


def safe_text(value):
    if value is None:
        return ""
    try:
        text = str(value)
    except:
        try:
            text = value.ToString()
        except:
            text = "<unprintable>"
    return text.replace("\r", " ").replace("\n", " ")


def safe_name(obj):
    try:
        return safe_text(obj.Name)
    except:
        return "<no name>"


def safe_category(obj):
    try:
        return safe_text(obj.DataModelObjectCategory)
    except:
        return "<unknown>"


def safe_object_id(obj):
    try:
        return safe_text(obj.ObjectId)
    except:
        return safe_category(obj) + "::" + safe_name(obj) + "::" + safe_text(id(obj))


def safe_children(obj):
    try:
        return list(obj.Children)
    except:
        return []


def get_parent(obj):
    try:
        return obj.Parent
    except:
        return None


def parent_chain(obj):
    chain = []
    cur = obj
    for _ in range(40):
        if cur is None:
            break
        chain.append(cur)
        cur = get_parent(cur)
    chain.reverse()
    return chain


def tree_path(obj):
    return " / ".join([safe_name(x) for x in parent_chain(obj)])


def owner_of(obj):
    chain = parent_chain(obj)
    for item in chain:
        cat = safe_category(item)
        if cat == "Analysis" or cat == "SolutionCombination":
            return item
    return None


def is_udr(obj):
    return safe_category(obj) == "UserDefinedResult"


def udr_expression(udr):
    try:
        return safe_text(udr.Expression).strip().upper()
    except:
        return ""


def collect_udrs(folder):
    found = []

    def walk(node):
        for child in safe_children(node):
            if is_udr(child):
                if udr_expression(child) in EXPRESSIONS:
                    found.append(child)
            else:
                walk(child)

    walk(folder)
    return found


def get_driver_label(udr):
    raw_values = []
    try:
        raw_values.append(safe_text(udr.PropertyByName("SetDriver").StringValue))
    except:
        pass
    try:
        raw_values.append(safe_text(udr.By))
    except:
        pass
    for raw in raw_values:
        if raw in SCREENING_DRIVER_TO_LABEL:
            return SCREENING_DRIVER_TO_LABEL[raw]
    nm = safe_name(udr)
    for label in SCREENING_LABELS:
        if label.lower() in nm.lower():
            return label
    return ""


def bolt_id_from_udr(udr):
    candidates = [safe_name(udr)]
    try:
        candidates.append(safe_name(udr.Location))
    except:
        pass
    for text in candidates:
        m = re.search(r"NS_BoltNode_([0-9]+)", text)
        if m:
            return m.group(1)
    return ""


def node_selection_name(udr):
    try:
        return safe_name(udr.Location)
    except:
        return ""


def result_unit(udr):
    try:
        return safe_text(udr.OutputUnit)
    except:
        return ""


def first_number(value):
    if value is None:
        return None
    try:
        return float(value)
    except:
        pass
    text = safe_text(value)
    m = re.search(r"[-+]?(?:\d+(?:\.\d*)?|\.\d+)(?:[Ee][-+]?\d+)?", text)
    if not m:
        return None
    try:
        return float(m.group(0))
    except:
        return None


def number_or_text(value):
    num = first_number(value)
    if num is None:
        return safe_text(value)
    if abs(num) < 1.0e-300:
        num = 0.0
    return num


def numeric_from_row(row, key):
    try:
        value = row["values"].get(key, "")
    except:
        return None
    if isinstance(value, float):
        return value
    return first_number(value)


def vector_magnitude(values):
    total = 0.0
    for value in values:
        if value is None:
            return ""
        total += value * value
    return math.sqrt(total)


def compute_derived_values(values):
    fx = first_number(values.get("ENFOX", ""))
    fy = first_number(values.get("ENFOY", ""))
    fz = first_number(values.get("ENFOZ", ""))
    mx = first_number(values.get("ENMOX", ""))
    my = first_number(values.get("ENMOY", ""))
    mz = first_number(values.get("ENMOZ", ""))
    return {
        "ForceTotal": vector_magnitude([fx, fy, fz]),
        "MomentTotal": vector_magnitude([mx, my, mz]),
        "ShearIfXAxial": vector_magnitude([fy, fz]),
        "ShearIfYAxial": vector_magnitude([fx, fz]),
        "ShearIfZAxial": vector_magnitude([fx, fy]),
        "MomentShearIfXAxial": vector_magnitude([my, mz]),
        "MomentShearIfYAxial": vector_magnitude([mx, mz]),
        "MomentShearIfZAxial": vector_magnitude([mx, my]),
    }


def derived_unit(quantity, unit_by_expr):
    if quantity.startswith("Moment"):
        return unit_by_expr.get("ENMOX", unit_by_expr.get("ENMOY", unit_by_expr.get("ENMOZ", "")))
    return unit_by_expr.get("ENFOX", unit_by_expr.get("ENFOY", unit_by_expr.get("ENFOZ", "")))


def fmt_value(value):
    if value is None:
        return ""
    if isinstance(value, float):
        if math.isnan(value) or math.isinf(value):
            return safe_text(value)
        return "{0:.12g}".format(value)
    return safe_text(value)


def approx_equal(a, b):
    if a is None or b is None:
        return True
    scale = max(1.0, abs(a), abs(b))
    return abs(a - b) <= 1.0e-9 * scale


def csv_escape(value):
    text = fmt_value(value)
    needs_quote = ("," in text) or ("\"" in text) or ("\n" in text) or ("\r" in text)
    text = text.replace("\"", "\"\"")
    if needs_quote:
        return "\"" + text + "\""
    return text


def csv_line(values):
    return ",".join([csv_escape(v) for v in values])


def ensure_dir(path):
    if not System.IO.Directory.Exists(path):
        System.IO.Directory.CreateDirectory(path)

def sanitize_file_name(text):
    text = safe_text(text).strip()
    bad = System.IO.Path.GetInvalidFileNameChars()
    for ch in bad:
        text = text.replace(safe_text(ch), "_")
    text = re.sub(r"\s+", "_", text)
    if not text:
        text = "unnamed"
    return text


def sanitize_sheet_name(text, used):
    text = safe_text(text).strip()
    for ch in [":", "\\", "/", "?", "*", "[", "]"]:
        text = text.replace(ch, "_")
    if not text:
        text = "Sheet"
    text = text[:31]
    base = text
    idx = 1
    while text.lower() in used:
        suffix = "_" + safe_text(idx)
        text = (base[:31 - len(suffix)] + suffix)
        idx += 1
    used.add(text.lower())
    return text


class LogBook(object):
    def __init__(self):
        self.lines = []
        self.warnings = 0

    def info(self, text):
        self.lines.append("[INFO] " + safe_text(text))

    def warn(self, text):
        self.warnings += 1
        self.lines.append("[WARN] " + safe_text(text))

    def error(self, text):
        self.lines.append("[ERROR] " + safe_text(text))


class BoltEnvironment(object):
    def __init__(self, owner):
        self.owner = owner
        self.owner_key = safe_object_id(owner)
        self.name = safe_name(owner)
        self.category = safe_category(owner)
        self.individual = {}
        self.screening = {}
        self.individual_root = None
        self.screening_root = None

    def add_individual(self, udr):
        expr = udr_expression(udr)
        bolt = bolt_id_from_udr(udr)
        if not expr or not bolt:
            return
        if bolt not in self.individual:
            self.individual[bolt] = {}
        self.individual[bolt][expr] = udr

    def add_screening(self, udr):
        expr = udr_expression(udr)
        label = get_driver_label(udr)
        if not expr or not label:
            return
        if expr not in self.screening:
            self.screening[expr] = {}
        self.screening[expr][label] = udr

    def all_udrs(self):
        objects = []
        for bolt in sorted(self.individual.keys()):
            for expr in EXPRESSIONS:
                if expr in self.individual[bolt]:
                    objects.append(self.individual[bolt][expr])
        for expr in EXPRESSIONS:
            if expr in self.screening:
                for label in SCREENING_LABELS:
                    if label in self.screening[expr]:
                        objects.append(self.screening[expr][label])
        return objects

    def individual_udr_count(self):
        count = 0
        for bolt in self.individual:
            count += len(self.individual[bolt].keys())
        return count

    def screening_udr_count(self):
        count = 0
        for expr in self.screening:
            count += len(self.screening[expr].keys())
        return count

    def missing_components(self):
        missing = []
        for bolt in sorted(self.individual.keys()):
            for expr in EXPRESSIONS:
                if expr not in self.individual[bolt]:
                    missing.append("Bolt {0}: {1}".format(bolt, expr))
        return missing

    def result_set_hint(self):
        if self.category == "SolutionCombination":
            try:
                count = int(self.owner.Definition.LoadCombinationCount)
                if count == 1:
                    return "1 load combination will be exported."
                return "{0} load combinations will be exported.".format(count)
            except Exception as exc:
                return "Load combination count is unavailable: " + safe_text(exc)
        try:
            data = self.owner.GetResultsData()
            try:
                if data is None:
                    return "Result-set list is unavailable; export will use the current result state."
                count = len(list(data.ListTimeFreq))
                if count == 0:
                    return "No result-set times were reported; export will use the current result state."
                if count == 1:
                    return "1 result set/time point will be exported."
                return "{0} result sets/time points will be exported when time history is enabled.".format(count)
            finally:
                try:
                    data.Dispose()
                except:
                    pass
        except Exception as exc:
            return "Result-set list is unavailable: " + safe_text(exc)

    def summary(self):
        bolt_count = len(self.individual.keys())
        ind_count = self.individual_udr_count()
        screen_count = self.screening_udr_count()
        return "{0} ({1}) - {2} bolts, {3} individual UDRs, {4} screening UDRs".format(
            self.name, self.category, bolt_count, ind_count, screen_count
        )

    def detail(self):
        lines = []
        lines.append("Tree object: " + self.name)
        lines.append("Type: " + self.category)
        lines.append("Individual bolt IDs: " + ", ".join(sorted(self.individual.keys())))
        lines.append("Individual UDRs: " + safe_text(self.individual_udr_count()))
        lines.append("Existing screening UDRs: " + safe_text(self.screening_udr_count()))
        lines.append("Result sets: " + self.result_set_hint())
        if self.screening_udr_count() == 0:
            lines.append("Screening availability: none found under '{0}'.".format(SCREENING_ROOT))
        else:
            lines.append("Screening availability: existing screening UDRs will be available as a separate table.")
        if self.individual_root is not None:
            lines.append("Individual root: " + tree_path(self.individual_root))
        if self.screening_root is not None:
            lines.append("Screening root: " + tree_path(self.screening_root))
        missing = self.missing_components()
        if missing:
            lines.append("")
            lines.append("Missing individual components:")
            lines.extend(missing[:12])
            if len(missing) > 12:
                lines.append("... {0} more".format(len(missing) - 12))
        else:
            lines.append("Completeness: every discovered bolt has all six ENFO/ENMO components.")
        return "\r\n".join(lines)


def discover_environments(log):
    envs_by_key = {}
    folders = DataModel.GetObjectsByType(DataModelObjectCategory.TreeGroupingFolder)
    for folder in folders:
        folder_name = safe_name(folder)
        if folder_name not in [INDIVIDUAL_ROOT, SCREENING_ROOT]:
            continue
        owner = owner_of(folder)
        if owner is None:
            continue
        key = safe_object_id(owner)
        if key not in envs_by_key:
            envs_by_key[key] = BoltEnvironment(owner)
        env = envs_by_key[key]
        udrs = collect_udrs(folder)
        if folder_name == INDIVIDUAL_ROOT:
            env.individual_root = folder
            for udr in udrs:
                env.add_individual(udr)
        elif folder_name == SCREENING_ROOT:
            env.screening_root = folder
            for udr in udrs:
                env.add_screening(udr)

    envs = []
    for key in envs_by_key:
        env = envs_by_key[key]
        if len(env.individual.keys()) > 0 or len(env.screening.keys()) > 0:
            envs.append(env)
    envs.sort(key=lambda e: (e.category, e.name))
    log.info("Discovered {0} bolt result environment(s).".format(len(envs)))
    return envs


def set_calculate_time_history(env, log):
    for udr in env.all_udrs():
        try:
            current = bool(udr.CalculateTimeHistory)
        except:
            current = False
        if current:
            continue
        try:
            udr.CalculateTimeHistory = True
            log.info("Set Calculate Time History = Yes on " + safe_name(udr))
        except Exception as exc:
            log.warn("Could not set Calculate Time History on {0}: {1}".format(safe_name(udr), safe_text(exc)))


def capture_state(udr):
    state = {}
    for attr in ["By", "SetNumber", "CombinationNumber", "Time", "DisplayTime"]:
        try:
            state[attr] = getattr(udr, attr)
        except:
            pass
    return state


def restore_state(udr, state):
    for attr in ["By", "SetNumber", "CombinationNumber", "Time", "DisplayTime"]:
        if attr not in state:
            continue
        try:
            setattr(udr, attr, state[attr])
        except:
            pass


def get_analysis_steps(env, log):
    steps = []
    if env.category == "SolutionCombination":
        count = 0
        try:
            count = int(env.owner.Definition.LoadCombinationCount)
        except Exception as exc:
            log.warn("Could not read load combination count for {0}: {1}".format(env.name, safe_text(exc)))
        if count < 1:
            count = 1
        for i in range(count):
            label = "Combination {0}".format(i)
            try:
                label = safe_text(env.owner.Definition.GetLoadCombinationName(i))
            except:
                pass
            steps.append({"kind": "combination", "index": i, "label": label, "time": ""})
        return steps

    times = []
    try:
        data = env.owner.GetResultsData()
        if data is not None:
            try:
                for t in data.ListTimeFreq:
                    times.append(t)
            finally:
                try:
                    data.Dispose()
                except:
                    pass
    except Exception as exc:
        log.warn("Could not read ListTimeFreq for {0}: {1}".format(env.name, safe_text(exc)))

    if times:
        for i, t in enumerate(times):
            steps.append({"kind": "set", "index": i + 1, "label": "Set {0}".format(i + 1), "time": number_or_text(t)})
    else:
        steps.append({"kind": "current", "index": 0, "label": "Current", "time": ""})
    return steps


def apply_step(udr, env, step, log):
    if env.category == "SolutionCombination":
        try:
            udr.CombinationNumber = int(step["index"])
            return
        except Exception as exc:
            log.warn("Could not set CombinationNumber={0} on {1}: {2}".format(step["index"], safe_name(udr), safe_text(exc)))
            return

    if step["kind"] == "current":
        return

    if SetDriverStyle is not None:
        try:
            udr.By = SetDriverStyle.ResultSet
        except:
            pass
    try:
        udr.SetNumber = int(step["index"])
    except:
        try:
            udr.SetNumber = int(step["index"]) - 1
        except Exception as exc:
            log.warn("Could not set result set {0} on {1}: {2}".format(step["index"], safe_name(udr), safe_text(exc)))


def evaluate_result(udr, log):
    try:
        udr.EvaluateAllResults()
    except Exception as exc:
        log.warn("Evaluation warning for {0}: {1}".format(safe_name(udr), safe_text(exc)))


def read_result_value(udr, log):
    min_val = None
    max_val = None
    try:
        min_val = first_number(udr.Minimum)
    except:
        pass
    try:
        max_val = first_number(udr.Maximum)
    except:
        pass
    if min_val is None and max_val is None:
        log.warn("No numeric Minimum/Maximum value found for " + safe_name(udr))
        return "", ""
    if min_val is None:
        return max_val, ""
    if max_val is None:
        return min_val, ""
    if approx_equal(min_val, max_val):
        return max_val, ""
    selected = max_val
    if abs(min_val) > abs(max_val):
        selected = min_val
    note = "min/max differ for {0}; selected larger absolute value (min={1}, max={2})".format(
        safe_name(udr), fmt_value(min_val), fmt_value(max_val)
    )
    log.warn(note)
    return selected, note


def extract_individual_rows(env, include_time_history, log):
    steps = get_analysis_steps(env, log)
    if not include_time_history and steps:
        steps = [steps[-1]]
    rows_by_bolt = {}
    unit_by_expr = {}
    states = {}
    for udr in env.all_udrs():
        states[safe_object_id(udr)] = (udr, capture_state(udr))

    try:
        for bolt in sorted(env.individual.keys()):
            rows = []
            expr_map = env.individual[bolt]
            for step in steps:
                row = {
                    "environment": env.name,
                    "environment_type": env.category,
                    "bolt": bolt,
                    "node_selection": "",
                    "step_index": step["index"],
                    "step_label": step["label"],
                    "time": step["time"],
                    "values": {},
                    "notes": [],
                }
                for expr in EXPRESSIONS:
                    if expr not in expr_map:
                        row["values"][expr] = ""
                        row["notes"].append("missing " + expr)
                        continue
                    udr = expr_map[expr]
                    if not row["node_selection"]:
                        row["node_selection"] = node_selection_name(udr)
                    if expr not in unit_by_expr:
                        unit_by_expr[expr] = result_unit(udr)
                    apply_step(udr, env, step, log)
                    evaluate_result(udr, log)
                    value, note = read_result_value(udr, log)
                    row["values"][expr] = value
                    if note:
                        row["notes"].append(note)
                derived = compute_derived_values(row["values"])
                for name in DERIVED_QUANTITIES:
                    row["values"][name] = derived.get(name, "")
                rows.append(row)
            rows_by_bolt[bolt] = rows
    finally:
        for key in states:
            obj, state = states[key]
            restore_state(obj, state)

    return rows_by_bolt, unit_by_expr


def compute_screening(rows_by_bolt):
    summaries = {}
    for bolt in rows_by_bolt:
        rows = rows_by_bolt[bolt]
        summary = []
        for expr in WIDE_VALUE_COLUMNS:
            values = []
            for row in rows:
                val = row["values"].get(expr, "")
                if isinstance(val, float):
                    values.append((val, row["time"], row["step_label"]))
                else:
                    num = first_number(val)
                    if num is not None:
                        values.append((num, row["time"], row["step_label"]))
            if not values:
                summary.append({
                    "expression": expr,
                    "max": "",
                    "time_of_max": "",
                    "min": "",
                    "time_of_min": "",
                    "max_step": "",
                    "min_step": "",
                    "sample_count": 0,
                })
                continue
            max_row = max(values, key=lambda x: x[0])
            min_row = min(values, key=lambda x: x[0])
            summary.append({
                "expression": expr,
                "max": max_row[0],
                "time_of_max": max_row[1],
                "max_step": max_row[2],
                "min": min_row[0],
                "time_of_min": min_row[1],
                "min_step": min_row[2],
                "sample_count": len(values),
            })
        summaries[bolt] = summary
    return summaries


def extract_existing_screening(env, log):
    rows = []
    for expr in EXPRESSIONS:
        if expr not in env.screening:
            continue
        for label in SCREENING_LABELS:
            if label not in env.screening[expr]:
                continue
            udr = env.screening[expr][label]
            evaluate_result(udr, log)
            min_val = ""
            max_val = ""
            try:
                min_val = number_or_text(udr.Minimum)
            except:
                pass
            try:
                max_val = number_or_text(udr.Maximum)
            except:
                pass
            rows.append({
                "environment": env.name,
                "expression": expr,
                "screening_result": label,
                "source_udr": safe_name(udr),
                "location": node_selection_name(udr),
                "minimum": min_val,
                "maximum": max_val,
                "unit": result_unit(udr),
            })
    return rows


def write_text_file(path, lines):
    writer = System.IO.StreamWriter(path, False, System.Text.Encoding.UTF8)
    try:
        for line in lines:
            writer.WriteLine(line)
    finally:
        writer.Close()

def quantity_type(quantity):
    if quantity in EXPRESSIONS:
        return "Raw UDR"
    if quantity in DERIVED_QUANTITIES:
        return "Derived"
    return "Other"


def quantity_unit(quantity, unit_by_expr):
    if quantity in EXPRESSIONS:
        return unit_by_expr.get(quantity, "")
    if quantity in DERIVED_QUANTITIES:
        return derived_unit(quantity, unit_by_expr)
    return ""

def set_cell(ws, row, col, value):
    if value is None:
        ws.Cells[row, col].Value2 = ""
    elif isinstance(value, bool):
        ws.Cells[row, col].Value2 = safe_text(value)
    elif isinstance(value, float) or isinstance(value, int):
        ws.Cells[row, col].Value2 = value
    else:
        ws.Cells[row, col].Value2 = fmt_value(value)


def write_table_to_sheet(ws, start_row, start_col, header, rows):
    r = start_row
    for c, value in enumerate(header):
        set_cell(ws, r, start_col + c, value)
    r += 1
    for row in rows:
        for c, value in enumerate(row):
            set_cell(ws, r, start_col + c, value)
        r += 1
    format_excel_table(ws, start_row, start_col, len(rows) + 1, len(header))
    for c, value in enumerate(header):
        if safe_text(value) in ["Notes", "Definition", "Value"]:
            try:
                ws.Columns[start_col + c].WrapText = True
            except:
                pass
    return r


def format_excel_table(ws, start_row, start_col, row_count, col_count):
    if row_count <= 0 or col_count <= 0:
        return
    try:
        header = ws.Range(ws.Cells[start_row, start_col], ws.Cells[start_row, start_col + col_count - 1])
        header.Font.Bold = True
        header.Interior.Color = 15132390
    except:
        pass
    try:
        full_range = ws.Range(ws.Cells[start_row, start_col], ws.Cells[start_row + row_count - 1, start_col + col_count - 1])
        full_range.AutoFilter()
    except:
        pass
    try:
        ws.Columns.AutoFit()
    except:
        pass


def freeze_top_row(ws):
    try:
        ws.Activate()
        ws.Application.ActiveWindow.SplitRow = 1
        ws.Application.ActiveWindow.FreezePanes = True
    except:
        pass


def format_sheet_finish(ws):
    try:
        ws.Columns.AutoFit()
    except:
        pass
    try:
        ws.Rows[1].Font.Bold = True
    except:
        pass
    freeze_top_row(ws)


def write_excel(path, env, rows_by_bolt, summaries, existing_screening, unit_by_expr, manifest_rows, log):
    excel_type = System.Type.GetTypeFromProgID("Excel.Application")
    if excel_type is None:
        raise Exception("Microsoft Excel COM automation is not available.")
    excel = System.Activator.CreateInstance(excel_type)
    excel.Visible = False
    excel.DisplayAlerts = False
    wb = None
    try:
        wb = excel.Workbooks.Add()
        while wb.Worksheets.Count > 1:
            wb.Worksheets[1].Delete()
        used_names = set()
        ws = wb.Worksheets[1]
        ws.Name = sanitize_sheet_name("Summary", used_names)

        summary_rows = []
        for bolt in sorted(summaries.keys()):
            for item in summaries[bolt]:
                summary_rows.append([
                    env.name,
                    bolt,
                    item["expression"],
                    quantity_type(item["expression"]),
                    quantity_unit(item["expression"], unit_by_expr),
                    item["max"],
                    item["time_of_max"],
                    item["max_step"],
                    item["min"],
                    item["time_of_min"],
                    item["min_step"],
                    item.get("sample_count", ""),
                ])
        write_table_to_sheet(ws, 1, 1, ["Environment", "Bolt", "Quantity", "QuantityType", "Unit", "MaximumOverTime", "TimeOfMaximum", "MaxStep", "MinimumOverTime", "TimeOfMinimum", "MinStep", "SampleCount"], summary_rows)
        format_sheet_finish(ws)

        ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
        ws.Name = sanitize_sheet_name("Tidy_Data", used_names)
        write_table_to_sheet(ws, 1, 1, ["Environment", "EnvironmentType", "Bolt", "NodeSelection", "StepIndex", "StepLabel", "Time", "Quantity", "QuantityType", "Value", "Unit", "Notes"], iter_tidy_rows(rows_by_bolt, unit_by_expr))
        format_sheet_finish(ws)

        ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
        ws.Name = sanitize_sheet_name("Wide_All_Bolts", used_names)
        write_table_to_sheet(ws, 1, 1, ["Environment", "EnvironmentType", "Bolt", "NodeSelection", "StepIndex", "StepLabel", "Time"] + WIDE_VALUE_COLUMNS + ["Notes"], iter_wide_rows(rows_by_bolt))
        format_sheet_finish(ws)

        ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
        ws.Name = sanitize_sheet_name("Units", used_names)
        unit_rows = []
        definitions = {
            "ForceTotal": "sqrt(ENFOX^2 + ENFOY^2 + ENFOZ^2)",
            "MomentTotal": "sqrt(ENMOX^2 + ENMOY^2 + ENMOZ^2)",
            "ShearIfXAxial": "sqrt(ENFOY^2 + ENFOZ^2); use only if X is axial for that bolt",
            "ShearIfYAxial": "sqrt(ENFOX^2 + ENFOZ^2); use only if Y is axial for that bolt",
            "ShearIfZAxial": "sqrt(ENFOX^2 + ENFOY^2); use only if Z is axial for that bolt",
            "MomentShearIfXAxial": "sqrt(ENMOY^2 + ENMOZ^2); use only if X is the relevant torsional axis",
            "MomentShearIfYAxial": "sqrt(ENMOX^2 + ENMOZ^2); use only if Y is the relevant torsional axis",
            "MomentShearIfZAxial": "sqrt(ENMOX^2 + ENMOY^2); use only if Z is the relevant torsional axis",
        }
        for quantity in WIDE_VALUE_COLUMNS:
            unit_rows.append([quantity, quantity_type(quantity), quantity_unit(quantity, unit_by_expr), definitions.get(quantity, "Direct Mechanical User Defined Result value")])
        write_table_to_sheet(ws, 1, 1, ["Quantity", "QuantityType", "Unit", "Definition"], unit_rows)
        format_sheet_finish(ws)

        for bolt in sorted(rows_by_bolt.keys()):
            ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
            ws.Name = sanitize_sheet_name("Bolt_" + bolt, used_names)
            hist_rows = []
            for row in rows_by_bolt[bolt]:
                hist = [env.name, env.category, bolt, row["node_selection"], row["step_index"], row["step_label"], row["time"]]
                for expr in WIDE_VALUE_COLUMNS:
                    hist.append(row["values"].get(expr, ""))
                hist.append("; ".join(row["notes"]))
                hist_rows.append(hist)
            next_row = write_table_to_sheet(ws, 1, 1, ["Environment", "EnvironmentType", "Bolt", "NodeSelection", "StepIndex", "StepLabel", "Time"] + WIDE_VALUE_COLUMNS + ["Notes"], hist_rows)
            next_row += 2
            screen_rows = []
            for item in summaries[bolt]:
                screen_rows.append([
                    item["expression"],
                    quantity_type(item["expression"]),
                    quantity_unit(item["expression"], unit_by_expr),
                    item["max"],
                    item["time_of_max"],
                    item["max_step"],
                    item["min"],
                    item["time_of_min"],
                    item["min_step"],
                    item.get("sample_count", ""),
                ])
            write_table_to_sheet(ws, next_row, 1, ["Quantity", "QuantityType", "Unit", "MaximumOverTime", "TimeOfMaximum", "MaxStep", "MinimumOverTime", "TimeOfMinimum", "MinStep", "SampleCount"], screen_rows)
            format_sheet_finish(ws)

        if existing_screening:
            ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
            ws.Name = sanitize_sheet_name("Existing_Screening", used_names)
            data_rows = []
            for row in existing_screening:
                data_rows.append([
                    row["environment"],
                    row["expression"],
                    row["screening_result"],
                    row["source_udr"],
                    row["location"],
                    row["minimum"],
                    row["maximum"],
                    row["unit"],
                ])
            write_table_to_sheet(ws, 1, 1, ["Environment", "Quantity", "ScreeningResult", "SourceUDR", "Location", "Minimum", "Maximum", "Unit"], data_rows)
            format_sheet_finish(ws)

        ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
        ws.Name = sanitize_sheet_name("Meta", used_names)
        write_table_to_sheet(ws, 1, 1, ["Field", "Value"], manifest_rows)
        format_sheet_finish(ws)

        ws = wb.Worksheets.Add(None, wb.Worksheets[wb.Worksheets.Count])
        ws.Name = sanitize_sheet_name("Log", used_names)
        for i, line in enumerate(log.lines):
            set_cell(ws, i + 1, 1, line)
        format_sheet_finish(ws)

        wb.SaveAs(path, 51)
    finally:
        if wb is not None:
            try:
                wb.Close(False)
            except:
                pass
        try:
            excel.Quit()
        except:
            pass


def default_output_folder():
    try:
        project_dir = safe_text(ExtAPI.DataModel.Project.ProjectDirectory)
        if project_dir and System.IO.Directory.Exists(project_dir):
            return System.IO.Path.Combine(project_dir, "bolt_result_exports")
    except:
        pass
    desktop = System.Environment.GetFolderPath(System.Environment.SpecialFolder.Desktop)
    return System.IO.Path.Combine(desktop, "bolt_result_exports")

        ok, reason = can_write_to_folder(folder)
        if not ok:
            MessageBox.Show("The output folder is not writable:\n\n{0}\n\nReason:\n{1}".format(folder, reason), "Bolt Export", MessageBoxButtons.OK, MessageBoxIcon.Warning)
            return
        env = self.environments[idx]
        missing = env.missing_components()
        if missing:
            preview = "\n".join(missing[:10])
            if len(missing) > 10:
                preview += "\n... {0} more".format(len(missing) - 10)
            answer = MessageBox.Show(
                "Some bolts are missing one or more ENFO/ENMO components.\n\n{0}\n\nExport anyway?".format(preview),
                "Bolt Export",
                MessageBoxButtons.YesNo,
                MessageBoxIcon.Warning,
            )
            if answer != DialogResult.Yes:
                return
        if bool(self.xlsx_radio.Checked) and not excel_is_available():
            MessageBox.Show(
                "Microsoft Excel COM automation is not available in this session. Select CSV output or install/repair Excel COM support.",
                "Bolt Export",
                MessageBoxButtons.OK,
                MessageBoxIcon.Warning,
            )
            self.csv_radio.Checked = True
            return
        self.selected_environment = env
        self.output_folder = folder
        self.output_excel = bool(self.xlsx_radio.Checked)
        self.include_time_history = bool(self.history_check.Checked)
        self.include_existing_screening = bool(self.screening_check.Checked)
        self.DialogResult = DialogResult.OK
        self.Close()

    def on_cancel(self, sender, args):
        self.DialogResult = DialogResult.Cancel
        self.Close()


def run_export(env, parent_folder, output_excel, include_time_history, include_existing_screening, log):
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    safe_env = sanitize_file_name(env.name)
    export_dir = System.IO.Path.Combine(parent_folder, "bolt_export_{0}_{1}".format(safe_env, timestamp))
    ensure_dir(export_dir)
    log.info("Output directory: " + export_dir)
    log.info("Selected environment: " + env.summary())

    set_calculate_time_history(env, log)
    rows_by_bolt, unit_by_expr = extract_individual_rows(env, include_time_history, log)
    summaries = compute_screening(rows_by_bolt)
    existing_screening = []
    if include_existing_screening:
        existing_screening = extract_existing_screening(env, log)
    manifest_rows = build_manifest_rows(
        env,
        export_dir,
        "Excel" if output_excel else "CSV",
        include_time_history,
        include_existing_screening,
        rows_by_bolt,
        summaries,
        existing_screening,
        unit_by_expr,
        log,
    )

    if output_excel:
        workbook = System.IO.Path.Combine(export_dir, "bolt_results_{0}.xlsx".format(safe_env))
        try:
            write_excel(workbook, env, rows_by_bolt, summaries, existing_screening, unit_by_expr, manifest_rows, log)
            log.info("Wrote Excel workbook: " + workbook)
        except Exception as exc:
            log.warn("Excel export failed; writing CSV fallback. Reason: " + safe_text(exc))
            output_excel = False
            manifest_rows = build_manifest_rows(
                env,
                export_dir,
                "CSV fallback after Excel failure",
                include_time_history,
                include_existing_screening,
                rows_by_bolt,
                summaries,
                existing_screening,
                unit_by_expr,
                log,
            )

    if not output_excel:
        for bolt in sorted(rows_by_bolt.keys()):
            path = System.IO.Path.Combine(export_dir, "bolt_{0}.csv".format(sanitize_file_name(bolt)))
            write_bolt_csv(path, env, bolt, rows_by_bolt[bolt], summaries.get(bolt, []), unit_by_expr)
            log.info("Wrote " + path)
        write_wide_all_csv(System.IO.Path.Combine(export_dir, "wide_all_bolts.csv"), rows_by_bolt)
        write_tidy_csv(System.IO.Path.Combine(export_dir, "tidy_bolt_results.csv"), rows_by_bolt, unit_by_expr)
        write_units_csv(System.IO.Path.Combine(export_dir, "units_and_definitions.csv"), unit_by_expr)
        write_summary_csv(System.IO.Path.Combine(export_dir, "computed_screening_summary.csv"), env, summaries, unit_by_expr)
        if include_existing_screening:
            write_existing_screening_csv(System.IO.Path.Combine(export_dir, "existing_screening_udr_values.csv"), existing_screening)
        write_manifest_csv(System.IO.Path.Combine(export_dir, "manifest.csv"), manifest_rows)

    write_log(System.IO.Path.Combine(export_dir, "export_log.txt"), log)
    time_rows = 0
    for bolt in rows_by_bolt:
        time_rows += len(rows_by_bolt[bolt])
    return export_dir, len(rows_by_bolt.keys()), time_rows, log.warnings


def main():
    log = LogBook()
    try:
        envs = discover_environments(log)
        if not envs:
            MessageBox.Show(
                "No bolt ENFO/ENMO result folders were found. Expected folders named 'Bolt Results' or 'Bolt Results - Max/Min Screening'.",
                "Bolt Export",
                MessageBoxButtons.OK,
                MessageBoxIcon.Warning,
            )
            return

        form = ExportOptionsForm(envs)
        if form.ShowDialog() != DialogResult.OK:
            return

        export_dir, bolt_count, time_rows, warnings = run_export(
            form.selected_environment,
            form.output_folder,
            form.output_excel,
            form.include_time_history,
            form.include_existing_screening,
            log,
        )

        MessageBox.Show(
            "Export complete.\n\nFolder:\n{0}\n\nBolts exported: {1}\nTime-history rows: {2}\nWarnings: {3}\n\nReview manifest and export_log for details.".format(export_dir, bolt_count, time_rows, warnings),
            "Bolt Export",
            MessageBoxButtons.OK,
            MessageBoxIcon.Information,
        )
    except Exception as exc:
        detail = traceback.format_exc()
        try:
            fallback = default_output_folder()
            ensure_dir(fallback)
            write_text_file(System.IO.Path.Combine(fallback, "bolt_export_error.txt"), [safe_text(detail)])
        except:
            pass
        MessageBox.Show(
            "Bolt export failed:\n\n{0}\n\nDetails:\n{1}".format(safe_text(exc), safe_text(detail)),
            "Bolt Export",
            MessageBoxButtons.OK,
            MessageBoxIcon.Error,
        )


main()
